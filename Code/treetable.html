É necessário o preenchimento dos campos "nodeId" e "parentNodeId" nos itens passados para a Tree Table.
HTML:
<table data-tree-table="" data-options="treeTable" class="tree-table">

<tr data-ng-repeat="item in treeTable.processedItens" data-tree-table-row="item">
JS:
angular.module('project', ['keeple.controls.tree-table'])

$rootScope.$on('treeTableReady', function () {
    $scope.treeTable.itens = itens;
});
$scope.treeTable = {
    itens: [],
    options: {
        lazyLoad: true
    },
    loadChildren: function (parentItem, callback) {
        parentItem.children = getChildren(parentItem.nodeId);
        callback(true);
        $scope.$apply();
    }
};
